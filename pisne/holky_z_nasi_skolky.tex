\beginsong{Holky z naší školky}[by={Maxim Turbulenc}]

\beginverse
\chordson
\[D]Majdalénka, \[G]Apolénka s \[A]Veronikou
\chordsoff
a taky \[D]Věrka, Zdenka, \[G]Majka, Lenka s \[A]Monikou,
no jasně \[D]Klára, Hančí, \[G]Bára, Mančí \[A]již nevím čí,
to všechno \[D]byly holky z \[G]naší školky -- \[A]senzační.
\endverse

\beginrefrain
\chordson
\[D]Jé, \[A]jé, \[G]jé, kdepak ty \[D]fajn holky \[G]jsou
a kde \[D]maj hračky \[G]svý, ty naše \[Emi7]lásky tříle\[A]tý.
\chordsoff
\[D]Pá, \[A]pá, \[G]pá, řekli sme \[D]pá před škol\[G]kou,
bylo \[D]nám právě \[G]šest a začla \[Emi7]další dívčí \[A]šou.
\endrefrain

\beginverse
\chordsoff
Ve škole Daniéla, Michaéla s Romanetou
a taky Adriána, Mariána se Žanetou
a hlavně príma Radka -- kamarádka, co všechno ví,
tyhlety holky byly naše víly školních dní.
\endverse

\beginrefrain
\chordsoff
Jé, jé, jé, kdepak ty fajn holky jsou
a kde maj žákovský naše lásky klukovský.
Čau, čau, čau, řekli jsme čau před školou,
táhlo nám na patnáct a začla další dívčí šou.
\endrefrain

\beginverse
\chordsoff
Na gymplu bezva Šárka, třída Klárka, Táňa jak sen
a taky senza Jenka v podkolénkách veselá jen
a všechny v sexy tričku -- postavičku měly ham-ham,
no prostě príma štace, inspirace k maturám.
\endverse

\beginrefrain
\chordsoff
Jé, jé, jé, kdepak ty fajn holky jsou
a kde maj zazděný naše lásky vysněný.
Au, au, au, vzlykli jsme au, čau a pá,
už se dál nekoná žádná dívčí školní šou.
\endrefrain

\beginverse
\chordsoff
I když pak poznali jsme spoustu dalších dívek a jmen,
plavovlásky, černovlásky, žár i sen,
v rytmu diska, z dálky -- z blízka, i v náručí,
přesto jsou stále holky z naší školky nejhezčí.
\endverse

\beginrefrain
\chordsoff
Jé, jé, jé, kdepak ty fajn holky jsou
a kde maj cůpky svý ty naše lásky tříletý.
\lrep\ Pá, pá, pá, říkame dál před školkou,
to se ví léta jdou, ale ty holky nestárnou. \rrep
\endrefrain
